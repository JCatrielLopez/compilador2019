int indice_if;
int indice_while;
int a[4];

begin

    indice_if := 3;
    if (indice_if > 0)
        begin
            print(indice_if);
        end;
    end_if;

    indice_while := 3;

    a[0] := 0;
    a[1] := 0;
    a[2] := 0;
    a[3] := 0;

    while (indice_while >= -1) do
        begin
            a[indice_while] := 2;
            print(indice_while);
            indice_while := indice_while - 1;
        end;;

    indice_if := a.first();
    print(indice_if);

    indice_if := a.length();
    print(indice_if);

    indice_if := a.last();
    print(indice_if);

end;
