int index_if;
int index_while;
int a[4];

begin

    index_if := 3;
    if (index_if > 0)
        begin
            print(index_if);
        end;
    end_if;

    index_while := 3;

    a[0] := 0;
    a[1] := 0;
    a[2] := 0;
    a[3] := 0;

    while (index_while >= -1) do
        begin
            a[index_while] := 2;
            print(index_while);
            index_while := index_while - 1;
        end;;

    index_if := a.first();
    print(index_if);

    index_if := a.length();
    print(index_if);

    index_if := a.last();
    print(index_if);

end;
